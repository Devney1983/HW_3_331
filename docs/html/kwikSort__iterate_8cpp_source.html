30/**
31 * @brief Partitions a subarray around a pivot for quicksort.
32 *
33 * @details
34 * The partition function selects a pivot value and reorders the subarray
35 * such that all elements less than the pivot are moved before it, and all
36 * elements greater than or equal to the pivot are moved after it.
37 *
38 * @param[in,out] set The vector of integers to partition.
39 * @param[in] start The starting index of the subarray to partition.
40 * @param[in] end The ending index of the subarray to partition.
41 *
42 * @pre `start` and `end` must be valid indices within the bounds of `set`.
43 * @post Elements in `set` are rearranged such that all elements before
44 * the pivot are smaller and all elements after are greater or equal.
45 *
46 * @return The index position of the pivot after partitioning.
47 *
48 *
49 * @see quickSort()
50 * @note Uses median-of-three pivot selection for improved performance.
51 */
52int partition(vector<int>& set, int start, int end)
53{
54 int pivotValue, pivotIndex, mid;
55
56 mid = (start + end) / 2;
57 swap(set[start], set[mid]);
58 pivotIndex = start;
59 pivotValue = set[start];
60
61 for (int scan = start + 1; scan <= end; scan++)
62 {
63 if (set[scan] < pivotValue)
64 {
65 pivotIndex++;
66 swap(set[pivotIndex], set[scan]);
67 }
68 }
69 swap(set[start], set[pivotIndex]);
70 return pivotIndex;
71}

73/**
74 * @brief Performs an iterative (non-recursive) quicksort on a vector.
75 *
76 * @details
77 * This function sorts a vector of integers in ascending order using
78 * the quicksort algorithm implemented with an explicit stack instead
79 * of recursive function calls.
80 *
81 * @param[in,out] set The vector of integers to be sorted.
82 * @param[in] start The starting index of the vector (typically 0).
83 * @param[in] end The ending index of the vector (typically `set.size() - 1`).
84 *
85 * @pre `start` and `end` must be valid indices in `set`, with `start <= end`.
86 * @post The vector `set` will be sorted in ascending order.
87 *
88 * @return void
89 *
90 *
91 * @see partition()
92 * @test Example:
93 * @code
94 * vector<int> nums = {10, 7, 8, 9, 1, 5};
95 * quickSort(nums, 0, nums.size() - 1);
96 * // nums is now {1, 5, 7, 8, 9, 10}
97 * @endcode
98 */
99void quickSort(vector<int>& set, int start, int end)
100{
101 stack<pair<int, int>> s;
102 s.push({start, end});
103
104 while (!s.empty())
105 {
106 int start = s.top().first;
107 int end = s.top().second;
108 s.pop();
109
110 if (start < end)
111 {
112 int p = partition(set, start, end);
113
114 s.push({start, p - 1});
115 s.push({p + 1, end});
116 }
117 }
118}
